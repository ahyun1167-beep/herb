        let currentQuiz = [];
        let currentQuestionIndex = 0;
        let score = 0;
        let selectedLevel = 0;

        const levelSelection = document.getElementById('level-selection');
        const quizScreen = document.getElementById('quiz-screen');
        const resultScreen = document.getElementById('result-screen');

        function startQuiz(level) {
            selectedLevel = level;
            currentQuiz = quizData[level];
            currentQuestionIndex = 0;
            score = 0;
            
            levelSelection.classList.add('hidden');
            resultScreen.classList.add('hidden');
            quizScreen.classList.remove('hidden');
            
            showQuestion();
        }

        function showQuestion() {
            resetState();
            const question = currentQuiz[currentQuestionIndex];
            document.getElementById('quiz-title').textContent = `🐴 ${selectedLevel}급 문제`;
            document.getElementById('progress-text').textContent = `${currentQuestionIndex + 1} / ${currentQuiz.length}`;
            document.getElementById('question-text').textContent = question.question;
        }

        function resetState() {
            document.getElementById('result-popup').classList.add('hidden');
            document.getElementById('explanation-box').classList.add('hidden');
            document.getElementById('answer-buttons').classList.remove('hidden');
        }

        function checkAnswer(userAnswer) {
            const question = currentQuiz[currentQuestionIndex];
            const isCorrect = userAnswer === question.answer;

            document.getElementById('answer-buttons').classList.add('hidden');
            document.getElementById('result-popup').classList.remove('hidden');
            const resultFeedback = document.getElementById('result-feedback');

            if (isCorrect) {
                score++;
                resultFeedback.textContent = "정답입니다! 🎉";
                resultFeedback.className = "p-4 rounded-lg mb-4 text-2xl font-bold bg-green-100 text-green-700";
            } else {
                resultFeedback.textContent = "오답입니다. 😥";
                resultFeedback.className = "p-4 rounded-lg mb-4 text-2xl font-bold bg-red-100 text-red-700";
                document.getElementById('explanation-box').classList.remove('hidden');
                document.getElementById('explanation-text').textContent = question.explanation;
            }

            if(currentQuestionIndex === currentQuiz.length - 1) {
                document.getElementById('next-button').textContent = "결과 보기";
            } else {
                document.getElementById('next-button').textContent = "다음 문제";
            }
        }

        function nextQuestion() {
            currentQuestionIndex++;
            if (currentQuestionIndex < currentQuiz.length) {
                showQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            quizScreen.classList.add('hidden');
            resultScreen.classList.remove('hidden');
            document.getElementById('score-text').textContent = `${currentQuiz.length}문제 중 ${score}개를 맞혔습니다!`;
            
            const percentage = score / currentQuiz.length;
            const trophy = document.getElementById('trophy');
            if (percentage === 1) {
                trophy.textContent = '🏆';
            } else if (percentage >= 0.7) {
                trophy.textContent = '🥇';
            } else if (percentage >= 0.4) {
                trophy.textContent = '🥈';
            } else {
                trophy.textContent = '🥉';
            }
        }

        function restart() {
            resultScreen.classList.add('hidden');
            levelSelection.classList.remove('hidden');
        }

    </script>
